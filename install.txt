添加文件 uwsgi.ini

[uwsgi]
# 网络配置
http=0.0.0.0:8000
# 项目路径
chdir=/home/lukejian/project/dashopt
home=/home/lukejian/project/dashopt/venv
# WSGI 配置 - 使用 module 方式（推荐）
module=dashopt.wsgi:application
# 如果使用 wsgi-file，应该是：
# wsgi-file = dashopt/wsgi.py
# 添加 Python 路径
pythonpath=/home/lukejian/project/dashopt
# 进程管理
master=true
processes=4
threads=2
# 文件配置
pidfile=%(chdir)/uwsgi.pid
daemonize=%(chdir)/uwsgi.log
# 日志
log-date=true
vacuum=true
die-on-term=true
# 环境变量
env=DJANGO_SETTINGS_MODULE=dashopt.settings



启动
uwsgi --ini uwsgi.ini
uwsgi --stop uwsgi.pid

添加venv之后
如果开发项目时没有设置venv，需要使用插件生成requirements.txt文件的内容

# 安装 pipreqs
pip install pipreqs
# 进入项目目录
cd D:\git\django\test
# 根据项目中的 import 语句自动生成 requirements.txt
pipreqs . --force
# 如果遇到编码问题
pipreqs . --force --encoding=utf-8
